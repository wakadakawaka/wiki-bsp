//  ВидОтчета			 - <СправочникСсылка.ВидыОтчетов>	 - вид отчета для определения счета ТР

//  Структура("ПоказателиИтога,ПоказателиТР", Соответствие(ПоказательОтчетаСсылка,Коэффициент), Структура("СКД,СКК"))
